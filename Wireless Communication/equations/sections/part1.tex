\section{Equations}
Forward and Inverse Transform, eq. \ref{eq:forward:inverse:1:transform} and eq. \ref{eq:forward:inverse:2:transform}.
\begin{equation}
  \label{eq:forward:inverse:1:transform}
  X(f) = \int_{-\infty}^{\infty} x(t) \, e^{-j2\pi f t} \, dt
\end{equation}

\begin{equation}
  \label{eq:forward:inverse:2:transform}
  x(t) = \int_{-\infty}^{\infty} X(f) \, e^{j2\pi f t} \, df
\end{equation}

Frequency Shift Property, eq. \ref{eq:freq:shift}.
\begin{equation}
  \label{eq:freq:shift}
  x(t) \, e^{j2\pi f_0 t} \quad \longleftrightarrow \quad X(f - f_0)
\end{equation}

Power, eq. \ref{eq:power}.
\begin{equation}
  \label{eq:power}
 P = \lim_{T \to \infty} \frac{1}{2T} \int_{-T}^{T} |x(t)|^2 \, dt
\end{equation}

Power in dBm, eq. \ref{eq:power:dbm}.
\begin{equation}
  \label{eq:power:dbm}
  P_{\text{dBm}} = 10 \log_{10}\left( \frac{P}{1\,\text{mW}} \right)
\end{equation}

Power ratios (dB), eq. \ref{eq:power:ratio}.
\begin{equation}
  \label{eq:power:ratio}
  G_{\text{dB}} = 10 \log_{10}\left( \frac{P_2}{P_1} \right)
\end{equation}

Absolute power (dBm), eq. \ref{eq:abs:power}.
\begin{equation}
  \label{eq:abs:power}
  P_{\text{dBm}} = 10 \log_{10}\left( \frac{P}{1\,\text{mW}} \right)
\end{equation}

Link budget, eq. \ref{eq:link:budget}.
\begin{equation}
  \label{eq:link:budget}
  P_{\text{rx}} = P_{\text{tx}} + G_{\text{tx}} + G_{\text{rx}} - L_{\text{path}} - L_{\text{other}}
\end{equation}

Additive White Gaussian Noise (AWGN), eq. \ref{eq:awgn}.
\begin{equation}
  \label{eq:awgn}
  y(t) = x(t) + n(t)
\end{equation}

$x(t)$ is the transmitted signal, $n(t)$ is the additivie noise, $y(t)$ is the received signal.

SNR, eq. \ref{eq:snr}.
\begin{equation}
  \label{eq:snr}
  \text{SNR} = \frac{P_s}{P_n}
\end{equation}

SNR in decibels (dB), eq. \ref{eq:snr:db}.
\begin{equation}
  \label{eq:snr:db}
  \text{SNR}_{\text{dB}} = 10 \log_{10}\left( \frac{P_s}{P_n} \right)
\end{equation}

Noise power spectral density, eq. \ref{eq:noise:power:spectral:density}.
\begin{equation}
  \label{eq:noise:power:spectral:density}
  N_0 = k_B T
\end{equation}

Total noise power over Bandwidth B, eq. \ref{eq:total:noise:power:b}.
\begin{equation}
  \label{eq:total:noise:power:b}
  P_n = k_B TB
\end{equation}

Noise floor (room temperature), eq. \ref{eq:noise:floor}.
\begin{equation}
  \label{eq:noise:floor}
  P_n \approx -174 + 10log_{10}(B) dBm
\end{equation}

Channel Capacity (Shannon Limit), eq. \ref{eq:channel:capacity}.
\begin{equation}
  \label{eq:channel:capacity}
  C = B log_2(1 + SNR)
\end{equation}

Message and carrier
\begin{equation}
  \label{eq:message}
  \begin{split}
  m(t): \text{ message signal}, \quad |m(t)| \leq 1\\
  c(t) = A_c \cos(2\pi f_c t)
  \end{split}
\end{equation}

Conventional AM signal:
\begin{equation}
  \label{eq:conv:am:signal}
  s(t) = A_c \left[1 + k_a m(t)\right] \cos(2\pi f_c t)
\end{equation}

DSB-SC signal model:
\begin{equation}
  \label{eq:dsb:sc}
  s(t) = c_t m(t)
\end{equation}

Required bandwidth (Single-Sideband Modulation (SSB)):
\begin{equation}
  \label{eq:bssb}
  B_{SSB} = B
\end{equation}

General angle-modulated signal
\begin{equation}
  \label{eq:gen:angle:mod:signal}
  s(t) = A_c \cos\left(2\pi f_c t + \phi(t)\right)
\end{equation}

Bandwidth (Carson's Rule):
\begin{equation}
  \label{eq:carsons:rule}
  B_{\text{FM}} \approx 2(\Delta f + B)
\end{equation}

Symbol duration:
\begin{equation}
  \label{eq:symbol:duration}
  T_s
\end{equation}

\begin{equation}
  \label{eq:ts}
  T_s = \frac{1}{R_s}
\end{equation}

Symbol rate:
\begin{equation}
  \label{eq:rs}
  R_s = \frac{1}{T_s}
\end{equation}

Bit rate:
\begin{equation}
  \label{eq:bit:rate}
  R_b = R_s log_2(M)
\end{equation}

Rule of thumb:
$$B \propto R_s$$

Spectral efficiency:
\begin{equation}
  \label{eq:spectral:efficiency}
  \eta = \frac{R_b}{B} \quad \text{(bits/s/Hz)}
\end{equation}

Cluster size:
\begin{equation}
  \label{eq:cluster:size}
  N = i^2 + ij + j^2
\end{equation}

Reuse distance:
\begin{equation}
  \label{eq:reuse:distance}
  \begin{split}
  D=\sqrt{3N}R\\
  D=R\sqrt{3N}
  \end{split}
\end{equation}

Feasibility of links:
\begin{equation}
  \label{eq:feasibility:links}
  P_r - (N + I) \geq \gamma_{min}
\end{equation}

Converting from W to dBm:
\begin{equation}
  \label{eq:watt:to:dbm}
  10log_{10}\left(\frac{P_{(W)}}{1}\right) = P_{(dBm)} - 30
\end{equation}
