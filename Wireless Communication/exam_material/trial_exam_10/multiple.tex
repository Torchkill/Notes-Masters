\section*{Section 1 --- Multiple Choice (20 questions)}
Choose the \textbf{one best answer} for each question.

\vspace{0.3cm}
\begin{multicols}{2}
\begin{enumerate}[leftmargin=*, itemsep=0.4em]
\item The unit dBm is a measure of:
\begin{enumerate}[label=\alph*)]
\item power referenced to 1 W
\item power referenced to 1 mW
\item voltage referenced to 1 V
\item path loss
\end{enumerate}

\item Thermal noise density at room temperature is approximately:
\begin{enumerate}[label=\alph*)]
\item $-174$ dBm/Hz
\item $-120$ dBm/Hz
\item $0$ dBm/Hz
\item $+10$ dBm/Hz
\end{enumerate}

\item If $T_s = 1$ ms, then the symbol rate $R_s$ is:
\begin{enumerate}[label=\alph*)]
\item $1{,}000$ symbols/s
\item $1$ symbol/s
\item $1{,}000{,}000$ symbols/s
\item $0.001$ symbols/s
\end{enumerate}

\item Intersymbol interference (ISI) happens when:
\begin{enumerate}[label=\alph*)]
\item symbols overlap in time at the receiver
\item the receiver has no thermal noise
\item antenna gain becomes infinite
\item the carrier frequency is exactly zero
\end{enumerate}

\item Spectral efficiency $\eta$ measures:
\begin{enumerate}[label=\alph*)]
\item data rate per unit bandwidth (bits/s/Hz)
\item noise power per unit bandwidth (dBm/Hz)
\item path loss per unit distance (dB/km)
\item antenna height per unit frequency
\end{enumerate}

\item Resource allocation aims to:
\begin{enumerate}[label=\alph*)]
\item maximize performance and minimize interference by distributing spectrum, time, and power
\item remove all multipath automatically
\item eliminate all noise sources from the receiver
\item keep transmit power fixed regardless of distance
\end{enumerate}

\item Which propagation mechanism allows a signal to bend around edges?
\begin{enumerate}[label=\alph*)]
\item Reflection
\item Diffraction
\item Refraction
\item Polarization
\end{enumerate}

\item Beamforming mainly provides:
\begin{enumerate}[label=\alph*)]
\item random delay
\item directional gain
\item lower bandwidth
\item no need for antennas
\end{enumerate}

\item In unlicensed bands, interference is typically:
\begin{enumerate}[label=\alph*)]
\item lower due to exclusive access
\item absent because of regulation
\item higher due to shared access
\item fixed and predictable
\end{enumerate}

\item Duplexing in wireless communication refers to:
\begin{enumerate}[label=\alph*)]
\item whether a system can send and receive data at the same time
\item how noise figure is calculated
\item how to compute free-space path loss
\item only how modulation shifts signals to higher frequencies
\end{enumerate}

\item SNR in dB is computed as:
\begin{enumerate}[label=\alph*)]
\item $10\log_{10}(P_s/P_n)$
\item $20\log_{10}(P_s/P_n)$
\item $10\log_{10}(P_n/P_s)$
\item $P_s/P_n$
\end{enumerate}

\item Which modulation carries the most bits/symbol?
\begin{enumerate}[label=\alph*)]
\item BPSK
\item QPSK
\item 16-QAM
\item all equal
\end{enumerate}

\item Free-space path loss increases when:
\begin{enumerate}[label=\alph*)]
\item frequency decreases
\item distance decreases
\item frequency or distance increases
\item transmit power increases
\end{enumerate}

\item A narrowband cellular IoT technology is:
\begin{enumerate}[label=\alph*)]
\item NB-IoT
\item Wi-Fi
\item GPS
\item FM radio
\end{enumerate}

\item Frequency reuse in cellular systems is used to:
\begin{enumerate}[label=\alph*)]
\item eliminate all interference
\item increase spatial reuse of spectrum
\item remove fading
\item replace coding
\end{enumerate}

\item Which item most directly improves link budget?
\begin{enumerate}[label=\alph*)]
\item larger packet header
\item higher antenna gain
\item longer cable
\item lower sampling rate
\end{enumerate}

\item A link is feasible when:
\begin{enumerate}[label=\alph*)]
\item $P_t > 0$ dBm
\item received SINR exceeds required threshold
\item carrier is below 1 GHz
\item antennas have equal gain
\end{enumerate}

\item Which statement best explains why media access control is hard in wireless networks?
\begin{enumerate}[label=\alph*)]
\item transmitters can always detect collisions on the shared medium
\item transmitters often cannot detect collisions
\item interference is eliminated by increasing bandwidth
\item noise is never present in wireless channels
\end{enumerate}

\item Which is an example of licensed spectrum usage?
\begin{enumerate}[label=\alph*)]
\item NB-IoT
\item Wi-Fi at 2.4 GHz
\item Bluetooth LE
\item Zigbee
\end{enumerate}

\item If transmit power and path losses are fixed, increasing bandwidth can lead to:
\begin{enumerate}[label=\alph*)]
\item higher noise power, reducing SNR (so capacity does not necessarily increase)
\item lower noise power, so SNR always increases
\item no change in noise power, so SNR stays constant
\item capacity becomes independent of bandwidth
\end{enumerate}

\end{enumerate}
\end{multicols}
